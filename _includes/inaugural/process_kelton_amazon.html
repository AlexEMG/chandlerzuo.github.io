def apply_mapping_logic(tbl_merged, output_data_dir):
    tbl_merged.fillna('0', inplace = True)
    tbl_merged['INCOME_BRACKET'] = tbl_merged.INCOME_BRACKET.astype(int)
    tbl_merged['DENSITY_BRACKET'] = tbl_merged.DENSITY_BRACKET.astype(int)
    tbl_merged['PREDICTED_SEGMENT'] = np.nan
    
    for logic_name in ['segment_subscription', 'segment_income', 'segment_density', 'subscription', 'income_density', 'income', 'density']:
        tbl_logic = pd.read_csv('{}/mapping_logic/mapping_{}.csv'.format(output_data_dir, logic_name),
                               dtype = {'SEGMENT': str, 'SUBSCRIPTION_PLAN_GROUP_ID': str,
                                        'INCOME_BRACKET': int, 'DENSITY_BRACKET': int})
        tbl_logic.columns = tbl_logic.columns.tolist()[:(tbl_logic.shape[1] - 1)] + ['PREDICT']
        join_keys = list(set(['SEGMENT', 'SUBSCRIPTION_PLAN_GROUP_ID', 'INCOME_BRACKET', 'DENSITY_BRACKET']).intersection(tbl_logic.columns.tolist()))
        print 'Processing ', logic_name, ' join keys ', join_keys
        tbl_logic = tbl_logic.iloc[1:, :] ## remove the first one
        if 'TOTAL' in tbl_logic.columns.tolist():
            tbl_logic.rename(columns = {'TOTAL':'SIZE'}, inplace = True)
        tbl_logic = tbl_logic.loc[tbl_logic.SIZE >= 10, join_keys + ['PREDICT']]
        print tbl_merged.columns.tolist()
        print tbl_logic.columns.tolist()
        print join_keys
        tbl_merged = pd.merge(tbl_merged, tbl_logic, how = 'left', on = join_keys)
        tbl_merged.loc[tbl_merged.PREDICTED_SEGMENT.isnull(), 'PREDICTED_SEGMENT'] = tbl_merged.loc[tbl_merged.PREDICTED_SEGMENT.isnull(), 'PREDICT']
        tbl_merged.drop(['PREDICT'], axis = 1, inplace = True)
    tbl_merged.loc[tbl_merged.PREDICTED_SEGMENT.isnull(), 'PREDICTED_SEGMENT'] = 'New Friend'
    return tbl_merged

def predict_amazon(target):
    if not os.path.isfile(file_amazon_retail_segment):
        merge_retail_segment_files()
        
    if not os.path.isfile(file_amazon_feature_merged):

        if not os.path.isfile(file_amazon_audible_merged):
            merge_audseg_amazon()
        tbl_merged = pd.read_csv(file_amazon_audible_merged)
        segment_col = 'AUDIBLE_SEGMENT'
    elif target == 'kelton':
        if not os.path.isfile(file_amazon_kelton_merged):
            merge_kelton_amazon()
        tbl_merged = pd.read_csv(file_amazon_kelton_merged)
        segment_col = 'KELTON_SEGMENT'

    tbl_true_predict = apply_mapping_logic(tbl_merged, output_data_dir)

    pd.pivot_table(tbl_true_predict, index = ['SEGMENT', 'SUBSCRIPTION_PLAN_GROUP_ID', 'INCOME_BRACKET', 'DENSITY_BRACKET'], values = 'CURRENT_CUSTOMER_ID', aggfunc = len).to_csv('{}/output/{}/summary_overall_prediction.csv'.format(data_dir, target))
